# 数据库设计

数据库名：`k8sgate`，字符集：`utf8mb4`，排序规则：`utf8mb4_general_ci`

## 1. users - 用户表

```sql
CREATE TABLE `users` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `username` varchar(64) NOT NULL COMMENT '登录名',
  `password_hash` varchar(255) NOT NULL COMMENT 'bcrypt加密密码',
  `display_name` varchar(128) NOT NULL DEFAULT '' COMMENT '显示名称',
  `email` varchar(255) NOT NULL DEFAULT '' COMMENT '邮箱',
  `role` enum('viewer','developer','admin') NOT NULL DEFAULT 'viewer' COMMENT '角色',
  `status` enum('active','disabled') NOT NULL DEFAULT 'active' COMMENT '账号状态',
  `password_changed_at` datetime DEFAULT NULL COMMENT '上次修改密码时间',
  `password_expires_at` datetime DEFAULT NULL COMMENT '密码过期时间',
  `failed_login_count` int(11) NOT NULL DEFAULT 0 COMMENT '连续登录失败次数',
  `locked_until` datetime DEFAULT NULL COMMENT '锁定截止时间',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`),
  KEY `idx_role` (`role`),
  KEY `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

## 2. projects - 项目表

```sql
CREATE TABLE `projects` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `name` varchar(128) NOT NULL COMMENT '项目名称',
  `description` varchar(512) NOT NULL DEFAULT '' COMMENT '项目描述',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_name` (`name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='项目表';
```

## 3. project_namespaces - 项目与命名空间映射表

```sql
CREATE TABLE `project_namespaces` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `project_id` bigint(20) unsigned NOT NULL COMMENT '项目ID',
  `namespace` varchar(253) NOT NULL COMMENT 'K8s命名空间名称',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_project_ns` (`project_id`, `namespace`),
  KEY `idx_namespace` (`namespace`),
  CONSTRAINT `fk_pn_project` FOREIGN KEY (`project_id`) REFERENCES `projects` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='项目与命名空间映射表';
```

## 4. user_projects - 用户与项目分配表

```sql
CREATE TABLE `user_projects` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `user_id` bigint(20) unsigned NOT NULL COMMENT '用户ID',
  `project_id` bigint(20) unsigned NOT NULL COMMENT '项目ID',
  `permission` enum('read','readwrite') NOT NULL DEFAULT 'read' COMMENT '权限级别',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_project` (`user_id`, `project_id`),
  KEY `idx_project_id` (`project_id`),
  CONSTRAINT `fk_up_user` FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) ON DELETE CASCADE,
  CONSTRAINT `fk_up_project` FOREIGN KEY (`project_id`) REFERENCES `projects` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户与项目分配表';
```

## 5. audit_logs - 操作审计日志表

```sql
CREATE TABLE `audit_logs` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `user_id` bigint(20) unsigned NOT NULL COMMENT '用户ID',
  `username` varchar(64) NOT NULL COMMENT '用户名（冗余字段，方便查询）',
  `action` varchar(16) NOT NULL COMMENT '操作类型：GET/POST/PUT/PATCH/DELETE',
  `resource_type` varchar(64) NOT NULL DEFAULT '' COMMENT '资源类型：Pod/Deployment/Service等',
  `resource_name` varchar(253) NOT NULL DEFAULT '' COMMENT '资源名称',
  `namespace` varchar(253) NOT NULL DEFAULT '' COMMENT '命名空间',
  `request_path` varchar(1024) NOT NULL COMMENT '原始请求路径',
  `status_code` int(11) NOT NULL DEFAULT 0 COMMENT 'HTTP响应状态码',
  `client_ip` varchar(45) NOT NULL DEFAULT '' COMMENT '客户端IP',
  `detail` varchar(1024) NOT NULL DEFAULT '' COMMENT '操作摘要',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_username` (`username`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_action` (`action`),
  KEY `idx_namespace` (`namespace`),
  KEY `idx_user_created` (`user_id`, `created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='操作审计日志表';
```

> 注意：audit_logs 表数据量会持续增长，建议按月分表或定期归档。初期可先用单表，数据量大时再做分表处理。

## 6. login_logs - 登录日志表

```sql
CREATE TABLE `login_logs` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `user_id` bigint(20) unsigned NOT NULL DEFAULT 0 COMMENT '用户ID（登录失败时可能为0）',
  `username` varchar(64) NOT NULL COMMENT '登录用户名',
  `client_ip` varchar(45) NOT NULL DEFAULT '' COMMENT '客户端IP',
  `result` enum('success','failed','locked') NOT NULL COMMENT '登录结果',
  `reason` varchar(255) NOT NULL DEFAULT '' COMMENT '失败原因',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_username` (`username`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_result` (`result`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='登录日志表';
```

## 7. ER 关系图

```
users 1───N user_projects N───1 projects 1───N project_namespaces
  |
  |── 1:N ── audit_logs
  |── 1:N ── login_logs
```

## 8. 数据量预估与维护策略

| 表 | 预估日增量 | 维护策略 |
|----|-----------|---------|
| users | 极少 | 无需特殊处理 |
| projects | 极少 | 无需特殊处理 |
| project_namespaces | 极少 | 无需特殊处理 |
| user_projects | 极少 | 无需特殊处理 |
| audit_logs | 约 5000~20000 条（500人日常操作） | 保留 6 个月，超期数据归档到备份表 |
| login_logs | 约 500~1500 条 | 保留 12 个月，超期数据清理 |
